# Example: Todo App (React)

A simple task manager app with authentication and realtime updates.

## Project Structure

```
my-todo-app/
  bool.config.json        # auto-generated by: bool link --project <id>
  .env.bool              # auto-generated (gitignored)
  bool/
    types.d.ts           # auto-generated types
    entities/
      tasks.json         # your schema
  src/
    lib/supabase.ts      # SDK client setup
    components/
      App.tsx
      SignInForm.tsx
      TaskList.tsx
      TaskForm.tsx
  package.json
```

## 1. Define Your Schema

**`bool/entities/tasks.json`:**
```json
{
  "type": "object",
  "properties": {
    "id": { "type": "string" },
    "user_id": { "type": "string" },
    "title": { "type": "string" },
    "done": { "type": "boolean" },
    "priority": {
      "type": "string",
      "enum": ["low", "medium", "high"]
    },
    "created_at": { "type": "string", "format": "date-time" }
  },
  "required": ["id", "user_id", "title", "done", "created_at"],
  "x-private": true
}
```

Push it:
```bash
npx bool entities push --dir bool/entities
```

## 3. Build the UI

**`src/components/App.tsx`:**
```tsx
import { BoolAuthProvider, AuthGate } from "bool-sdk/react";
import { SignInForm } from "./SignInForm";
import { Dashboard } from "./Dashboard";

export default function App() {
  return (
    <BoolAuthProvider>
      <AuthGate fallback={<SignInForm />}>
        <Dashboard />
      </AuthGate>
    </BoolAuthProvider>
  );
}
```

**`src/components/SignInForm.tsx`:**
```tsx
import { useState } from "react";
import { bool } from "../lib/supabase";

export function SignInForm() {
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");

  const handleSignUp = async () => {
    setLoading(true);
    setError("");
    try {
      await bool.auth.signUp({ email, password });
    } catch (err) {
      setError(err.message);
    }
    setLoading(false);
  };

  const handleSignIn = async () => {
    setLoading(true);
    setError("");
    try {
      await bool.auth.signInWithPassword({ email, password });
    } catch (err) {
      setError(err.message);
    }
    setLoading(false);
  };

  return (
    <div style={{ maxWidth: 400, margin: "0 auto", padding: 20 }}>
      <h1>Todo App</h1>
      <input
        type="email"
        placeholder="Email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
      />
      <input
        type="password"
        placeholder="Password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
      />
      <button onClick={handleSignIn} disabled={loading}>
        Sign In
      </button>
      <button onClick={handleSignUp} disabled={loading}>
        Sign Up
      </button>
      {error && <p style={{ color: "red" }}>{error}</p>}
    </div>
  );
}
```

**`src/components/Dashboard.tsx`:**
```tsx
import { useState, useEffect } from "react";
import { useBoolAuth } from "bool-sdk/react";
import { bool } from "../lib/supabase";
import { TaskList } from "./TaskList";
import { TaskForm } from "./TaskForm";

export function Dashboard() {
  const { user, signOut } = useBoolAuth();
  const [tasks, setTasks] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Load tasks
    const loadTasks = async () => {
      const data = await bool.entities.tasks.list("-created_at");
      setTasks(data);
      setLoading(false);
    };
    loadTasks();

    // Subscribe to changes
    const unsubscribe = bool.subscribeToChanges("tasks", async () => {
      const updated = await bool.entities.tasks.list("-created_at");
      setTasks(updated);
    });

    return unsubscribe;
  }, []);

  const handleAddTask = async (title: string, priority: string) => {
    const newTask = await bool.entities.tasks.create({
      title,
      priority,
      done: false,
      created_at: new Date().toISOString(),
    });
    setTasks([newTask, ...tasks]);
  };

  const handleToggleDone = async (id: string, done: boolean) => {
    await bool.entities.tasks.update(id, { done: !done });
    setTasks(tasks.map((t) => (t.id === id ? { ...t, done: !done } : t)));
  };

  const handleDeleteTask = async (id: string) => {
    await bool.entities.tasks.delete(id);
    setTasks(tasks.filter((t) => t.id !== id));
  };

  if (loading) return <p>Loading...</p>;

  return (
    <div style={{ maxWidth: 600, margin: "0 auto", padding: 20 }}>
      <div style={{ display: "flex", justifyContent: "space-between" }}>
        <h1>Tasks</h1>
        <div>
          <p>{user?.email}</p>
          <button onClick={signOut}>Sign Out</button>
        </div>
      </div>

      <TaskForm onAdd={handleAddTask} />
      <TaskList
        tasks={tasks}
        onToggleDone={handleToggleDone}
        onDelete={handleDeleteTask}
      />
    </div>
  );
}
```

**`src/components/TaskForm.tsx`:**
```tsx
import { useState } from "react";

export function TaskForm({ onAdd }) {
  const [title, setTitle] = useState("");
  const [priority, setPriority] = useState("medium");

  const handleSubmit = () => {
    if (!title.trim()) return;
    onAdd(title, priority);
    setTitle("");
    setPriority("medium");
  };

  return (
    <div style={{ marginBottom: 20 }}>
      <input
        type="text"
        placeholder="What needs to be done?"
        value={title}
        onChange={(e) => setTitle(e.target.value)}
        onKeyPress={(e) => e.key === "Enter" && handleSubmit()}
      />
      <select value={priority} onChange={(e) => setPriority(e.target.value)}>
        <option value="low">Low</option>
        <option value="medium">Medium</option>
        <option value="high">High</option>
      </select>
      <button onClick={handleSubmit}>Add Task</button>
    </div>
  );
}
```

**`src/components/TaskList.tsx`:**
```tsx
export function TaskList({ tasks, onToggleDone, onDelete }) {
  return (
    <ul style={{ listStyle: "none", padding: 0 }}>
      {tasks.map((task) => (
        <li
          key={task.id}
          style={{
            padding: 10,
            border: "1px solid #ccc",
            marginBottom: 10,
            borderRadius: 4,
          }}
        >
          <div style={{ display: "flex", gap: 10, alignItems: "center" }}>
            <input
              type="checkbox"
              checked={task.done}
              onChange={() => onToggleDone(task.id, task.done)}
            />
            <span
              style={{
                textDecoration: task.done ? "line-through" : "none",
                flex: 1,
              }}
            >
              {task.title}
            </span>
            <span style={{ fontSize: 12, color: "#666" }}>
              {task.priority}
            </span>
            <button onClick={() => onDelete(task.id)}>Delete</button>
          </div>
        </li>
      ))}
    </ul>
  );
}
```

## 4. Deploy

```bash
npx bool deploy
```

Your app is live at `https://<slug>.bool.so`.

## What's Happening

1. **Auth**: Users sign up/in via Bool. Each user gets a session.
2. **Data**: `bool.entities.tasks.list()` returns only the current user's tasks (RLS).
3. **Realtime**: Subscriptions via `bool.subscribeToChanges()` broadcast pings. Refetch on ping.
4. **Types**: `bool/types.d.ts` makes everything type-safe.

No backend code, no database setup, no deployment scripts. Just your app and the Bool SDK.
